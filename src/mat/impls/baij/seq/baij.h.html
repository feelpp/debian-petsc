217: /*
218:   PetscKernel_A_gets_A_times_B_2: A = A * B with size bs=2

220:   Input Parameters:
221: +  A,B - square bs by bs arrays stored in column major order
222: -  W   - bs*bs work arrary

224:   Output Parameter:
225: .  A = A * B
226: */

228: #define PetscKernel_A_gets_A_times_B_2(A,B,W) 0; \
229:   { \
230:     PetscMemcpy(W,A,4*sizeof(MatScalar)); \
231:     A[0] = W[0]*B[0] + W[2]*B[1]; \
232:     A[1] = W[1]*B[0] + W[3]*B[1]; \
233:     A[2] = W[0]*B[2] + W[2]*B[3]; \
234:     A[3] = W[1]*B[2] + W[3]*B[3]; \
235:   }

237: /*
238:   PetscKernel_A_gets_A_minus_B_times_C_2: A = A - B * C with size bs=2
