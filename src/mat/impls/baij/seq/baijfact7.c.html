239: {

244:   const PetscInt *r,*ic;
245:   PetscInt       i,j,k,nz,nzL,row;
246:   const PetscInt n=a->mbs,*ai=a->i,*aj=a->j,*bi=b->i,*bj=b->j;

248:   MatScalar      *rtmp,*pc,*mwork,*v,*pv,*aa=a->a;

250:   PetscReal      shift = info->shiftamount;

253:   ISGetIndices(isrow,&r);

256:   /* generate work space needed by the factorization */
257:   PetscMalloc2(bs2*n,MatScalar,&rtmp,bs2,MatScalar,&mwork);
258:   PetscMemzero(rtmp,bs2*n*sizeof(MatScalar));

260:   for (i=0; i<n; i++) {
261:     /* zero rtmp */
262:     /* L part */
263:     nz    = bi[i+1] - bi[i];
264:     bjtmp = bj + bi[i];
265:     for  (j=0; j<nz; j++) {
266:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
267:     }

269:     /* U part */

276:     /* load in initial (unfactored row) */

573: PetscErrorCode MatLUFactorNumeric_SeqBAIJ_6_NaturalOrdering(Mat B,Mat A,const MatFactorInfo *info)
574: {
575:   Mat            C =B;
576:   Mat_SeqBAIJ    *a=(Mat_SeqBAIJ*)A->data,*b=(Mat_SeqBAIJ*)C->data;
578:   PetscInt       i,j,k,nz,nzL,row;
579:   const PetscInt n=a->mbs,*ai=a->i,*aj=a->j,*bi=b->i,*bj=b->j;
580:   const PetscInt *ajtmp,*bjtmp,*bdiag=b->diag,*pj,bs2=a->bs2;
581:   MatScalar      *rtmp,*pc,*mwork,*v,*pv,*aa=a->a;
582:   PetscInt       flg;
583:   PetscReal      shift = info->shiftamount;

586:   /* generate work space needed by the factorization */
587:   PetscMalloc2(bs2*n,MatScalar,&rtmp,bs2,MatScalar,&mwork);
588:   PetscMemzero(rtmp,bs2*n*sizeof(MatScalar));

590:   for (i=0; i<n; i++) {
591:     /* zero rtmp */
592:     /* L part */
593:     nz    = bi[i+1] - bi[i];
594:     bjtmp = bj + bi[i];
595:     for  (j=0; j<nz; j++) {
596:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
597:     }

599:     /* U part */
600:     nz    = bdiag[i] - bdiag[i+1];
601:     bjtmp = bj + bdiag[i+1]+1;
602:     for  (j=0; j<nz; j++) {
603:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
604:     }

606:     /* load in initial (unfactored row) */
607:     nz    = ai[i+1] - ai[i];
608:     ajtmp = aj + ai[i];
609:     v     = aa + bs2*ai[i];
610:     for (j=0; j<nz; j++) {
611:       PetscMemcpy(rtmp+bs2*ajtmp[j],v+bs2*j,bs2*sizeof(MatScalar));
612:     }

614:     /* elimination */
615:     bjtmp = bj + bi[i];
616:     nzL   = bi[i+1] - bi[i];
617:     for (k=0; k < nzL; k++) {
618:       row = bjtmp[k];
619:       pc  = rtmp + bs2*row;
620:       for (flg=0,j=0; j<bs2; j++) {
621:         if (pc[j]!=0.0) {
622:           flg = 1;
623:           break;
624:         }
625:       }
626:       if (flg) {
627:         pv = b->a + bs2*bdiag[row];
628:         /* PetscKernel_A_gets_A_times_B(bs,pc,pv,mwork); *pc = *pc * (*pv); */
629:         PetscKernel_A_gets_A_times_B_6(pc,pv,mwork);

631:         pj = b->j + bdiag[row+1]+1; /* begining of U(row,:) */
632:         pv = b->a + bs2*(bdiag[row+1]+1);
633:         nz = bdiag[row] - bdiag[row+1] - 1; /* num of entries inU(row,:), excluding diag */
634:         for (j=0; j<nz; j++) {
635:           /* PetscKernel_A_gets_A_minus_B_times_C(bs,rtmp+bs2*pj[j],pc,pv+bs2*j); */
636:           /* rtmp+bs2*pj[j] = rtmp+bs2*pj[j] - (*pc)*(pv+bs2*j) */
637:           v    = rtmp + bs2*pj[j];
638:           PetscKernel_A_gets_A_minus_B_times_C_6(v,pc,pv);
639:           pv  += bs2;
640:         }
641:         PetscLogFlops(432*nz+396); /* flops = 2*bs^3*nz + 2*bs^3 - bs2) */
642:       }
643:     }

645:     /* finished row so stick it into b->a */
646:     /* L part */
647:     pv = b->a + bs2*bi[i];
648:     pj = b->j + bi[i];
649:     nz = bi[i+1] - bi[i];
650:     for (j=0; j<nz; j++) {
651:       PetscMemcpy(pv+bs2*j,rtmp+bs2*pj[j],bs2*sizeof(MatScalar));
652:     }

654:     /* Mark diagonal and invert diagonal for simplier triangular solves */
655:     pv   = b->a + bs2*bdiag[i];
656:     pj   = b->j + bdiag[i];
657:     PetscMemcpy(pv,rtmp+bs2*pj[0],bs2*sizeof(MatScalar));
658:     /* PetscKernel_A_gets_inverse_A(bs,pv,v_pivots,v_work); */
659:     PetscKernel_A_gets_inverse_A_6(pv,shift);

661:     /* U part */
662:     pv = b->a + bs2*(bdiag[i+1]+1);
663:     pj = b->j + bdiag[i+1]+1;
664:     nz = bdiag[i] - bdiag[i+1] - 1;
665:     for (j=0; j<nz; j++) {
666:       PetscMemcpy(pv+bs2*j,rtmp+bs2*pj[j],bs2*sizeof(MatScalar));
667:     }
668:   }
669:   PetscFree2(rtmp,mwork);

671:   C->ops->solve          = MatSolve_SeqBAIJ_6_NaturalOrdering;
672:   C->ops->solvetranspose = MatSolveTranspose_SeqBAIJ_6_NaturalOrdering;
673:   C->assembled           = PETSC_TRUE;

675:   PetscLogFlops(1.333333333333*6*6*6*n); /* from inverting diagonal blocks */
676:   return(0);
677: }
