215: PetscErrorCode MatLUFactorNumeric_SeqBAIJ_5(Mat B,Mat A,const MatFactorInfo *info)
216: {
217:   Mat            C     =B;
218:   Mat_SeqBAIJ    *a    =(Mat_SeqBAIJ*)A->data,*b=(Mat_SeqBAIJ*)C->data;
219:   IS             isrow = b->row,isicol = b->icol;
221:   const PetscInt *r,*ic;
222:   PetscInt       i,j,k,nz,nzL,row;
223:   const PetscInt n=a->mbs,*ai=a->i,*aj=a->j,*bi=b->i,*bj=b->j;
224:   const PetscInt *ajtmp,*bjtmp,*bdiag=b->diag,*pj,bs2=a->bs2;
225:   MatScalar      *rtmp,*pc,*mwork,*v,*pv,*aa=a->a,work[25];
226:   PetscInt       flg,ipvt[5];
227:   PetscReal      shift = info->shiftamount;

230:   ISGetIndices(isrow,&r);
231:   ISGetIndices(isicol,&ic);

233:   /* generate work space needed by the factorization */
234:   PetscMalloc2(bs2*n,MatScalar,&rtmp,bs2,MatScalar,&mwork);
235:   PetscMemzero(rtmp,bs2*n*sizeof(MatScalar));

237:   for (i=0; i<n; i++) {
238:     /* zero rtmp */
239:     /* L part */
240:     nz    = bi[i+1] - bi[i];
241:     bjtmp = bj + bi[i];
242:     for  (j=0; j<nz; j++) {
243:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
244:     }

246:     /* U part */
247:     nz    = bdiag[i] - bdiag[i+1];
248:     bjtmp = bj + bdiag[i+1]+1;
249:     for  (j=0; j<nz; j++) {
250:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
251:     }

253:     /* load in initial (unfactored row) */
254:     nz    = ai[r[i]+1] - ai[r[i]];
255:     ajtmp = aj + ai[r[i]];
256:     v     = aa + bs2*ai[r[i]];
257:     for (j=0; j<nz; j++) {
258:       PetscMemcpy(rtmp+bs2*ic[ajtmp[j]],v+bs2*j,bs2*sizeof(MatScalar));
259:     }

261:     /* elimination */
262:     bjtmp = bj + bi[i];
263:     nzL   = bi[i+1] - bi[i];
264:     for (k=0; k < nzL; k++) {
265:       row = bjtmp[k];
266:       pc  = rtmp + bs2*row;
267:       for (flg=0,j=0; j<bs2; j++) {
268:         if (pc[j]!=0.0) {
269:           flg = 1;
270:           break;
271:         }
272:       }
273:       if (flg) {
274:         pv = b->a + bs2*bdiag[row];
275:         /* PetscKernel_A_gets_A_times_B(bs,pc,pv,mwork); *pc = *pc * (*pv); */
276:         PetscKernel_A_gets_A_times_B_5(pc,pv,mwork);

278:         pj = b->j + bdiag[row+1]+1; /* begining of U(row,:) */
279:         pv = b->a + bs2*(bdiag[row+1]+1);
280:         nz = bdiag[row] - bdiag[row+1] - 1; /* num of entries inU(row,:), excluding diag */
281:         for (j=0; j<nz; j++) {
282:           /* PetscKernel_A_gets_A_minus_B_times_C(bs,rtmp+bs2*pj[j],pc,pv+bs2*j); */
283:           /* rtmp+bs2*pj[j] = rtmp+bs2*pj[j] - (*pc)*(pv+bs2*j) */
284:           v    = rtmp + bs2*pj[j];
285:           PetscKernel_A_gets_A_minus_B_times_C_5(v,pc,pv);
286:           pv  += bs2;
287:         }
288:         PetscLogFlops(250*nz+225); /* flops = 2*bs^3*nz + 2*bs^3 - bs2) */
289:       }
290:     }

292:     /* finished row so stick it into b->a */
293:     /* L part */
294:     pv = b->a + bs2*bi[i];
295:     pj = b->j + bi[i];
296:     nz = bi[i+1] - bi[i];
297:     for (j=0; j<nz; j++) {
298:       PetscMemcpy(pv+bs2*j,rtmp+bs2*pj[j],bs2*sizeof(MatScalar));
299:     }

301:     /* Mark diagonal and invert diagonal for simplier triangular solves */
302:     pv   = b->a + bs2*bdiag[i];
303:     pj   = b->j + bdiag[i];
304:     PetscMemcpy(pv,rtmp+bs2*pj[0],bs2*sizeof(MatScalar));
305:     /* PetscKernel_A_gets_inverse_A(bs,pv,v_pivots,v_work); */
306:     PetscKernel_A_gets_inverse_A_5(pv,ipvt,work,shift);

308:     /* U part */
309:     pv = b->a + bs2*(bdiag[i+1]+1);
310:     pj = b->j + bdiag[i+1]+1;
311:     nz = bdiag[i] - bdiag[i+1] - 1;
312:     for (j=0; j<nz; j++) {
313:       PetscMemcpy(pv+bs2*j,rtmp+bs2*pj[j],bs2*sizeof(MatScalar));
314:     }
315:   }

317:   PetscFree2(rtmp,mwork);
318:   ISRestoreIndices(isicol,&ic);
319:   ISRestoreIndices(isrow,&r);

321:   C->ops->solve          = MatSolve_SeqBAIJ_5;
322:   C->ops->solvetranspose = MatSolveTranspose_SeqBAIJ_5;
323:   C->assembled           = PETSC_TRUE;

325:   PetscLogFlops(1.333333333333*5*5*5*n); /* from inverting diagonal blocks */
326:   return(0);
327: }

504: PetscErrorCode MatLUFactorNumeric_SeqBAIJ_5_NaturalOrdering(Mat B,Mat A,const MatFactorInfo *info)
505: {
506:   Mat            C =B;
507:   Mat_SeqBAIJ    *a=(Mat_SeqBAIJ*)A->data,*b=(Mat_SeqBAIJ*)C->data;
509:   PetscInt       i,j,k,nz,nzL,row;
510:   const PetscInt n=a->mbs,*ai=a->i,*aj=a->j,*bi=b->i,*bj=b->j;
511:   const PetscInt *ajtmp,*bjtmp,*bdiag=b->diag,*pj,bs2=a->bs2;
512:   MatScalar      *rtmp,*pc,*mwork,*v,*vv,*pv,*aa=a->a,work[25];
513:   PetscInt       flg,ipvt[5];
514:   PetscReal      shift = info->shiftamount;

517:   /* generate work space needed by the factorization */
518:   PetscMalloc2(bs2*n,MatScalar,&rtmp,bs2,MatScalar,&mwork);
519:   PetscMemzero(rtmp,bs2*n*sizeof(MatScalar));

521:   for (i=0; i<n; i++) {
522:     /* zero rtmp */
523:     /* L part */
524:     nz    = bi[i+1] - bi[i];
525:     bjtmp = bj + bi[i];
526:     for  (j=0; j<nz; j++) {
527:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
528:     }

530:     /* U part */
531:     nz    = bdiag[i] - bdiag[i+1];
532:     bjtmp = bj + bdiag[i+1]+1;
533:     for  (j=0; j<nz; j++) {
534:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
535:     }

537:     /* load in initial (unfactored row) */
538:     nz    = ai[i+1] - ai[i];
539:     ajtmp = aj + ai[i];
540:     v     = aa + bs2*ai[i];
541:     for (j=0; j<nz; j++) {
542:       PetscMemcpy(rtmp+bs2*ajtmp[j],v+bs2*j,bs2*sizeof(MatScalar));
543:     }

545:     /* elimination */
546:     bjtmp = bj + bi[i];
547:     nzL   = bi[i+1] - bi[i];
548:     for (k=0; k < nzL; k++) {
549:       row = bjtmp[k];
550:       pc  = rtmp + bs2*row;
551:       for (flg=0,j=0; j<bs2; j++) {
552:         if (pc[j]!=0.0) {
553:           flg = 1;
554:           break;
555:         }
556:       }
557:       if (flg) {
558:         pv = b->a + bs2*bdiag[row];
559:         /* PetscKernel_A_gets_A_times_B(bs,pc,pv,mwork); *pc = *pc * (*pv); */
560:         PetscKernel_A_gets_A_times_B_5(pc,pv,mwork);

562:         pj = b->j + bdiag[row+1]+1; /* begining of U(row,:) */
563:         pv = b->a + bs2*(bdiag[row+1]+1);
564:         nz = bdiag[row] - bdiag[row+1] - 1; /* num of entries inU(row,:), excluding diag */
565:         for (j=0; j<nz; j++) {
566:           /* PetscKernel_A_gets_A_minus_B_times_C(bs,rtmp+bs2*pj[j],pc,pv+bs2*j); */
567:           /* rtmp+bs2*pj[j] = rtmp+bs2*pj[j] - (*pc)*(pv+bs2*j) */
568:           vv   = rtmp + bs2*pj[j];
569:           PetscKernel_A_gets_A_minus_B_times_C_5(vv,pc,pv);
570:           pv  += bs2;
571:         }
572:         PetscLogFlops(250*nz+225); /* flops = 2*bs^3*nz + 2*bs^3 - bs2) */
573:       }
574:     }

576:     /* finished row so stick it into b->a */
577:     /* L part */
578:     pv = b->a + bs2*bi[i];
579:     pj = b->j + bi[i];
580:     nz = bi[i+1] - bi[i];
581:     for (j=0; j<nz; j++) {
582:       PetscMemcpy(pv+bs2*j,rtmp+bs2*pj[j],bs2*sizeof(MatScalar));
583:     }

585:     /* Mark diagonal and invert diagonal for simplier triangular solves */
586:     pv   = b->a + bs2*bdiag[i];
587:     pj   = b->j + bdiag[i];
588:     PetscMemcpy(pv,rtmp+bs2*pj[0],bs2*sizeof(MatScalar));
589:     /* PetscKernel_A_gets_inverse_A(bs,pv,v_pivots,v_work); */
590:     PetscKernel_A_gets_inverse_A_5(pv,ipvt,work,shift);

592:     /* U part */
593:     pv = b->a + bs2*(bdiag[i+1]+1);
594:     pj = b->j + bdiag[i+1]+1;
595:     nz = bdiag[i] - bdiag[i+1] - 1;
596:     for (j=0; j<nz; j++) {
597:       PetscMemcpy(pv+bs2*j,rtmp+bs2*pj[j],bs2*sizeof(MatScalar));
598:     }
599:   }
600:   PetscFree2(rtmp,mwork);

602:   C->ops->solve          = MatSolve_SeqBAIJ_5_NaturalOrdering;
603:   C->ops->solvetranspose = MatSolveTranspose_SeqBAIJ_5_NaturalOrdering;
604:   C->assembled           = PETSC_TRUE;

606:   PetscLogFlops(1.333333333333*5*5*5*n); /* from inverting diagonal blocks */
607:   return(0);
