 31:   /* for each row k */

 34:     /*initialize k-th row with elements nonzero in row k of A */
 35:     jmin = ai[k]; jmax = ai[k+1];
 36:     if (jmin < jmax) {
 37:       ap = aa + jmin*25;
 38:       for (j = jmin; j < jmax; j++) {
 39:         vj       = aj[j];   /* block col. index */
 40:         rtmp_ptr = rtmp + vj*25;
 41:         for (i=0; i<25; i++) *rtmp_ptr++ = *ap++;
 42:       }
