 26: /*@C
 27:   MatFinalizePackage - This function destroys everything in the Petsc interface to the Mat package. It is
 28:   called from PetscFinalize().

 30:   Level: developer

 32: .keywords: Petsc, destroy, package, mathematica
 33: .seealso: PetscFinalize()
 34: @*/
 35: PetscErrorCode  MatFinalizePackage(void)
 36: {
 37:   MatBaseName    nnames,names = MatBaseNameList;

 41:   while (names) {
 42:     nnames = names->next;
 43:     PetscFree(names->bname);
 44:     PetscFree(names->sname);
 45:     PetscFree(names->mname);
 46:     PetscFree(names);
 47:     names  = nnames;
 48:   }
 49:   PetscFunctionListDestroy(&MatList);
 50:   PetscFunctionListDestroy(&MatOrderingList);
 51:   PetscFunctionListDestroy(&MatColoringList);
 52:   PetscFunctionListDestroy(&MatPartitioningList);
 53:   PetscFunctionListDestroy(&MatCoarsenList);
 54:   MatBaseNameList                  = NULL;
 55:   MatPackageInitialized            = PETSC_FALSE;
 56:   MatRegisterAllCalled             = PETSC_FALSE;
 57:   MatOrderingRegisterAllCalled     = PETSC_FALSE;
 58:   MatColoringRegisterAllCalled     = PETSC_FALSE;
 59:   MatPartitioningRegisterAllCalled = PETSC_FALSE;
 60:   MatCoarsenRegisterAllCalled      = PETSC_FALSE;
 61:   return(0);
 62: }
