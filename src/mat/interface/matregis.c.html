 55: /*
 56:     This is used by MatSetType() to make sure that at least one
 57:     MatRegisterAll() is called. In general, if there is more than one
 58:     DLL, then MatRegisterAll() may be called several times.
 59: */
 60: extern PetscBool MatRegisterAllCalled;

 64: /*@C
 65:   MatRegisterAll - Registers all of the matrix types in PETSc

 67:   Not Collective

 69:   Level: advanced

 71: .keywords: KSP, register, all

 73: .seealso:  MatRegisterDestroy()
 74: @*/
 75: PetscErrorCode  MatRegisterAll(void)
 76: {

 80:   MatRegisterAllCalled = PETSC_TRUE;

 82:   MatRegister(MATMFFD,           MatCreate_MFFD);

 84:   MatRegister(MATMPIMAIJ,        MatCreate_MAIJ);
 85:   MatRegister(MATSEQMAIJ,        MatCreate_MAIJ);
 86:   MatRegister(MATMAIJ,           MatCreate_MAIJ);

 88:   MatRegister(MATIS,             MatCreate_IS);
 89:   MatRegister(MATSHELL,          MatCreate_Shell);
 90:   MatRegister(MATCOMPOSITE,      MatCreate_Composite);

 92:   MatRegisterBaseName(MATAIJ,MATSEQAIJ,MATMPIAIJ);
 93:   MatRegister(MATMPIAIJ,         MatCreate_MPIAIJ);
 94:   MatRegister(MATSEQAIJ,         MatCreate_SeqAIJ);

 96:   MatRegisterBaseName(MATAIJPERM,MATSEQAIJPERM,MATMPIAIJPERM);
 97:   MatRegister(MATMPIAIJPERM,     MatCreate_MPIAIJPERM);
 98:   MatRegister(MATSEQAIJPERM,     MatCreate_SeqAIJPERM);

100:   MatRegisterBaseName(MATAIJCRL,MATSEQAIJCRL,MATMPIAIJCRL);
101:   MatRegister(MATSEQAIJCRL,      MatCreate_SeqAIJCRL);
102:   MatRegister(MATMPIAIJCRL,      MatCreate_MPIAIJCRL);

104:   MatRegisterBaseName(MATBAIJ,MATSEQBAIJ,MATMPIBAIJ);
105:   MatRegister(MATMPIBAIJ,        MatCreate_MPIBAIJ);
106:   MatRegister(MATSEQBAIJ,        MatCreate_SeqBAIJ);

108:   MatRegisterBaseName(MATSBAIJ,MATSEQSBAIJ,MATMPISBAIJ);
109:   MatRegister(MATMPISBAIJ,       MatCreate_MPISBAIJ);
110:   MatRegister(MATSEQSBAIJ,       MatCreate_SeqSBAIJ);

112:   MatRegisterBaseName(MATBSTRM,MATSEQBSTRM,MATMPIBSTRM);
113:   MatRegister(MATMPIBSTRM,       MatCreate_MPIBSTRM);
114:   MatRegister(MATSEQBSTRM,       MatCreate_SeqBSTRM);

116:   MatRegisterBaseName(MATSBSTRM,MATSEQSBSTRM,MATMPISBSTRM);
117:   MatRegister(MATMPISBSTRM,      MatCreate_MPISBSTRM);
118:   MatRegister(MATSEQSBSTRM,      MatCreate_SeqSBSTRM);

120:   MatRegisterBaseName(MATDENSE,MATSEQDENSE,MATMPIDENSE);
121:   MatRegister(MATMPIDENSE,       MatCreate_MPIDense);
122:   MatRegister(MATSEQDENSE,       MatCreate_SeqDense);

124:   MatRegister(MATMPIADJ,         MatCreate_MPIAdj);
125:   MatRegister(MATSCATTER,        MatCreate_Scatter);
126:   MatRegister(MATBLOCKMAT,       MatCreate_BlockMat);
127:   MatRegister(MATNEST,           MatCreate_Nest);


130: #if defined PETSC_HAVE_CUSP
131:   MatRegisterBaseName(MATAIJCUSP,MATSEQAIJCUSP,MATMPIAIJCUSP);
132:   MatRegister(MATSEQAIJCUSP,     MatCreate_SeqAIJCUSP);
133:   MatRegister(MATMPIAIJCUSP,     MatCreate_MPIAIJCUSP);
134: #endif

136: #if defined PETSC_HAVE_TXPETSCGPU
137:   MatRegisterBaseName(MATAIJCUSPARSE,MATSEQAIJCUSPARSE,MATMPIAIJCUSPARSE);
138:   MatRegister(MATSEQAIJCUSPARSE, MatCreate_SeqAIJCUSPARSE);
139:   MatRegister(MATMPIAIJCUSPARSE, MatCreate_MPIAIJCUSPARSE);
140: #endif

142: #if defined PETSC_HAVE_FFTW
143:   MatRegister(MATFFTW,           MatCreate_FFTW);
144: #endif
145: #if defined PETSC_HAVE_ELEMENTAL
146:   MatRegister(MATELEMENTAL,      MatCreate_Elemental);
147: #endif
148:   return(0);
149: }
