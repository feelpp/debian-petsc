  4:        program main
  5: #include <finclude/petscsys.h>
  6:        PetscErrorCode  ierr
  7:        PetscMPIInt     rank
  8:        character*(80) arch

 10:        call PetscInitialize(PETSC_NULL_CHARACTER,ierr)

 12:        if (ierr .ne. 0) then
 13:          print*, 'Unable to begin PETSc program'
 14:        endif

 16:        call MPI_Comm_rank(PETSC_COMM_WORLD,rank,ierr)
 17:        print*, 'Greetings from', rank

 19:        call PetscGetArchType(arch,ierr)
 20:        print*, 'Machine type ',arch

 22:        call PetscFinalize(ierr)
 23:        end
