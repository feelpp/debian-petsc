  7:       subroutine FortranCopy(n,x,y)
  8:       implicit none
  9:       PetscScalar  x(*),y(*)
 10:       PetscInt n
 11:       PetscInt i
 12:       PETSC_AssertAlignx(16,x(1))
 13:       PETSC_AssertAlignx(16,y(1))
 14:       do 10,i=1,n
 15:         y(i) = x(i)
 16:  10   continue
 17:       return
 18:       end
